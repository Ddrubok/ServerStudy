int32 sum = 0;

mutex m;

Spinlock spinlock;

void Add()
{
	for (int32 i = 0; i < 100000; i++)
	{
		lock_guard<Spinlock> guard(spinlock);
		sum++;
	}
}

void Sub()
{
	for (int32 i = 0; i < 100000; i++)
	{
		lock_guard<Spinlock> guard(spinlock);
		sum--;
	}
}

{
	thread t1(Add);
	thread t2(Sub);

	t1.join();
	t2.join();
